Envoyer email notification

Si il y a plusieurs email en copie, il faut les séparer par point virgule [;].

CONTROLEUR

Ajouter le factory notificationEmail par le système d'injection d'AngularJS dans votre contrôleur pour l'utiliser.

Vous pourrez par la suite vous inspirer du code ci-dessous.

                        
                    
        //Initialisation des variables
        $scope.email = {};
        $scope.email.a = "";
        $scope.email.cc = "";
        $scope.email.objet = "";
        $scope.email.message = "";
        /*Envoi email*/
        $scope.envoyerEmailNotification = function (data_email)
        {
            //data_email représente les données issues du view
            $rootScope.modalLoading(true);  //Afficher loading
            var dataObj = {
                a: data_email.a,
                cc: data_email.cc,
                objet: data_email.objet,
                message: data_email.message,
            }
            notificationEmail.envoyer(dataObj).then(function (datas) {
                $rootScope.modalLoading(false); //Fermer loading
            });
        }

Note:

L'URL qui est utilisé pour l'envoie des notifications se trouve dans "Gestion menu & variables" sous le nom de Notification.
Chiffre en lettre

Pour utiliser l'outil "chiffre en lettre", appeler la fonction javascript NumberToLetter.
                        
                    
        //Chiffre en lettre
        $scope.cl = {};
        $scope.cl.chiffre = 0;
        $scope.cl.lettre = "";
        $scope.chiffreEnLettre = function(nombre)
        {
           var lettre = NumberToLetter(nombre.chiffre);
           $scope.cl.lettre = lettre;
        }
